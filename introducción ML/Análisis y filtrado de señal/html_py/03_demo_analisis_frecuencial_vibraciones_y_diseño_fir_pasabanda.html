  • An√°lisis frecuencial de las se√±ales mediante FFT
  • Dise√±o de un filtro FIR pasabanda en la zona de frecuencias de los cojinetes
  • Trazado de la respuesta en frecuencia del filtro FIR dise√±ado
  • Filtrado de las se√±ales mediante el filtro FIR
  • An√°lisis frecuencial de la se√±al filtrada mediante FFT

# DISE√ëO DE UN FILTRO FIR PASABAJOS
# Obtencion de coeficientes del filtro 
wc1 = 1500/(fm/2);                              # Frecuencia de corte (normalizada 0=0, 1=frec_shannon)
wc2 = 2200/(fm/2);                              # Frecuencia de corte (normalizada 0=0, 1=frec_shannon)
b = sig.firwin(251,[wc1,wc2],window='hamming',pass_zero=False);     # Coeficientes del fir (defecto: ventana de Hamming)



# Trazado de la respuesta en frecuencia del filtro fir
plt.figure(2,figsize=(15,5)); 
plt.semilogy(f,abs(np.fft.fft(b,N)));  
plt.grid('on')
plt.title(u'Respuesta en frecuencia del filtro FIR (escala logar√≠tmica)');




# FILTRADO DE LAS SE√ëALES
axf = sig.lfilter(b,1,ax);
ayf = sig.lfilter(b,1,ay);



plt.figure(3,figsize=(15,10));

# Importamos PyAudio para reproducir se√±ales como audio
import pyaudio

# --------------------------------------------------
# SONIFICACI√ìN DE LA SE√ëAL ORIGINAL
# --------------------------------------------------

# Se√±al de aceleraci√≥n original (a_x)
chunk = ax

# Inicializamos el sistema de audio
p = pyaudio.PyAudio()

# Abrimos un stream de salida (mono, float32, fs = 5000 Hz)
stream = p.open(format=pyaudio.paFloat32,
                channels=1,
                rate=5000,
                output=True,
                frames_per_buffer=1)

# Reproducimos la se√±al original
stream.write(chunk.astype(np.float32).tobytes())

# Cerramos el stream y liberamos recursos
stream.close()
p.terminate()
