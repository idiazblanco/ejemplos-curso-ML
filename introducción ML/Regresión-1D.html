import numpy as np
import matplotlib.pyplot as plt

Generaci√≥n de datos para el ejemplo¬∂

Generamos un conjunto de $N$ datos $(x,y)$, donde $x\in \mathbb R^1, y\in \mathbb R^1$ linealmente relacionados $$ y = a x + b + \epsilon $$ con una incertidumbre $\epsilon$ que puede representar ruido en el sensor $y$

In¬†[¬†]:
# N√∫mero de datos del ejemplo
N = 1000;

# Coeficientes del modelo
a = 1; b = 1;

# variables de entrada o regresores
X = np.random.randn(N,1);

# errores o ruido de sensor
epsilon = 0.5*np.random.randn(N,1);

# variable de respuesta generada con el modelo (incluyendo ruido)
y = a*X + b + epsilon;

Entrenamiento del modelo¬∂

In¬†[¬†]:
# importamos la clase Ridge
from sklearn.linear_model import Ridge

# definimos un modelo lineal para entrenamiento con Ridge regression
model = Ridge(alpha=0.00001)

# ajustamos el modelo
model.fit(X,y)

# podemos obtener los coeficientes del modelo
print('valores estimados: ')
print(f'coeficientes = {model.coef_}')
print(f't√©rmino independiente = {model.intercept_}')

Inferencia del modelo con datos de test¬∂

In¬†[¬†]:
# datos de test
Xtest = np.linspace(-5,5,1000)[:,np.newaxis];

# predicci√≥n
ytest = model.predict(Xtest)

# visualizaci√≥n gr√°fica de los resultados
plt.figure(1,figsize=(7,7))
plt.plot(X[:,0],y,'.');
plt.plot(Xtest,ytest,'r')
plt.axis('equal');
plt.grid('on');
