Aproximaci√≥n modelos no lineales como combinaci√≥n lineal de t√©rminos no lineales $$ y = f(x,\theta) \approx a_1 f_1(x) + a_2 f_2(x) + \dots a_n f_n(x) $$ donde $f$ es una funci√≥n no lineal desconocida y $f_i$ son t√©rminos no lineales que se pueden elegir convenientemente en cada problema, aprovechando el conocimiento disponible.

In¬†[¬†]:
import matplotlib.pyplot as plt
import numpy as np

from scipy.signal import square
from scipy.signal import sawtooth
In¬†[¬†]:
# definimos la funci√≥n "verdadera" (ground truth)
f = lambda x: square(x) + 0.5*sawtooth(2*x) + 0.2*x + 0.02*x**2

# N√∫mero de datos del ejemplo
N = 3000;

# Generamos datos acordes con el modelo polin√≥mico
a3, a2, a1, a0 = 1, -5, 1, 2
epsilon = 0.3*np.random.randn(N);

x = np.linspace(0,20,N);
y = f(x) + epsilon; 

plt.plot(x,y)

Aproximamos nuestra funci√≥n por una base de funciones no lineales $$ x,\quad x^2,\quad \sin(x),\quad \cos(x),\quad \sin(2x),\quad \cos(2x),\quad \sin(3x),\quad \cos(3x),\quad \sin(4x),\quad \cos(4x) $$

In¬†[¬†]:
# creamos una matriz con los t√©rminos no lineales del polinomio
X = np.column_stack((x, x**2, np.sin(x), np.cos(x), np.sin(2*x), np.cos(2*x), np.sin(3*x), np.cos(3*x), np.sin(4*x), np.cos(4*x)))
In¬†[¬†]:
from sklearn.linear_model import Ridge
modelo = Ridge(alpha=0.0001)

modelo.fit(X,y)

print(f'coeficientes          = {modelo.coef_}')
print(f't√©rmino independiente = {modelo.intercept_}')

# generamos datos de test nuevos
xtest = np.linspace(-10,30,1000)
ytest = f(xtest)

# construimos la matriz X con los mismos t√©rminos no lineales que para el entrenamiento
Xtest = np.column_stack((xtest, xtest**2, np.sin(xtest), np.cos(xtest), np.sin(2*xtest), np.cos(2*xtest), np.sin(3*xtest), np.cos(3*xtest), np.sin(4*xtest), np.cos(4*xtest)))
ypred = modelo.predict(Xtest)

plt.figure(2,figsize=(7,7));
plt.plot(x,y,'.',alpha=0.3);			# datos de entrenamiento
